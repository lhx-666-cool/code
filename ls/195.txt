#include<iostream>

using namespace std;

const int N = 1e3;

int g[N][N], f[N][N];
int n;

void zf() {
    for (int i = 0; i < n; i++) {
        f[i][i] = 1;
    }
}
void dc() {
    for (int i = 0; i < n; i++) {
        for (int j = i + 1; j < n; j++) {
            f[i][j] = f[i][j] | f[j][i]; // 将a[i][j]和a[j][i]做或运算
            f[j][i] = f[i][j]; // 将结果赋给a[j][i]
        }
    }
}
void cd() {
    for (int k = 0; k < n; k++) {
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                f[i][j] = f[i][j] | (f[i][k] & f[k][j]); // 将a[i][j]和a[i][k]&a[k][j]做或运算
            }
        }
    }
}

int main() {
	ios::sync_with_stdio(false);
	cin.tie(nullptr);
	cin >> n;
	for (int i = 0; i < n; i ++) {
		for (int j = 0; j < n; j ++) {
			cin >> g[i][j];
			f[i][j] = g[i][j];
		}
	}
	zf(), dc(), cd();
	bool st = true;
	for (int i = 0; i < n; i ++) {
		for (int j = 0; j < n; j ++) {
			if (f[i][j] != g[i][j]) {
				st = false;
				goto res;
			}
 		}
	}
	res:;
	if (st) {
		cout << "yes";
	}else {
		cout << "no";
	}
}
