#include <iostream>

using namespace std;

const int N = 110;

int f[N][N];
int n;
int st[N];

void dfs(int u) {
	st[u] = true;
	for (int i = 0; i < n; i ++) 
		if (!st[i] && f[u][i] == 1) 
			dfs(i);
}

int main() {
	ios::sync_with_stdio(false);
	cin.tie(nullptr);

	cin >> n;

	for (int i = 0; i < n; i ++)
		for (int j = 0; j < n; j ++)
			cin >> f[i][j];

	bool res = true;

	dfs(0);

	for (int i = 0; i < n; i ++)
		if (!st[i])
			res = false;
	if (res)
		cout << "Y" << endl;
	else
		cout << "N" << endl;
}
