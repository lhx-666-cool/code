#include<iostream>
using namespace std;
int main() {
    int n, m;
    cin >> n >> m;
    for (int i = n + 1; i < m; i ++) {
        for (int j = i + 1; j < m; j ++) {
            for (int k = j + 1; k < m; k ++) {
                cout << "<" << i << "," << j << "," << k << ">" << endl;
            }
        }
    }
}
