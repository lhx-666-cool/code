#include <iostream>

using namespace std;

const int N = 510;

int f[N][N];

int main() {
	ios::sync_with_stdio(false);
	cin.tie(nullptr);
	int n, m;
	cin >> n >> m;
	for (int i = 0; i < n; i ++) 
		for (int j = 0; j < m; j ++)
			cin >> f[j][i];

	for (int i = 0; i < m ; i ++) {
		for (int j = 0; j < n; j ++)
			cout << f[i][j] << " ";
		cout << endl;
	}
}
