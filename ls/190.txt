#include<iostream>
using namespace std;

const int N = 1e3 + 10;
int jz[N][N];
int n;

void cd() {
    for (long long k = 0; k < n; k++) {
        for (long long i = 0; i < n; i++) {
            for (long long j = 0; j < n; j++) {
                jz[i][j] = jz[i][j] | (jz[i][k] & jz[k][j]); // 将a[i][j]和a[i][k]&a[k][j]做或运算
            }
        }
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cin >> n;
    for (int i = 0; i < n; i ++) {
        for (int j = 0; j < n; j ++) {
            cin >> jz[i][j];
        }
    }
    cd();
    for (int i = 0; i < n; i ++) {
        for (int j = 0; j < n; j ++) {
            cout << jz[i][j] << " ";
        }
        cout << endl;
    }
    return 0;
}
