#include<iostream>
#include<cctype>
#include<string>
#include<vector>
#include<set>
using namespace std;
const int N = 1e5 + 10;
typedef pair<int, string>IS;
string inp[N];
vector<string> res;
int h = -1;
string temp;
set<IS> s;
void dfs(int c, int j, int cnt) {
    if (c == res.size()) {
        s.emplace(cnt, temp);
        return;
    }
    for (int i = j; i < res.size(); i ++) {
        string old = temp;
        if (!temp.empty()) {
            temp += ",";
        }
        temp += res[i];
        dfs(c + 1, i + 1, cnt + 1);
        temp = old;
    }
    dfs(c + 1, j, cnt);
}
int main() {
    char str[1100];
    scanf("%s", str);
    if (isdigit(str[0])) {
        scanf("%s", str);
    }
    for (int i = 0; str[i] != '}'; i ++) {
        if (str[i] == '{' || str[i] == ',') {
            h ++;
        }else {
            inp[h] += str[i];
        }
    }
    for (int i = 0; i <= h; i ++) {
        for (int j = 0; j <= h; j ++) {
            if (!inp[i].empty()) {
                res.push_back("<" + inp[i] + "," + inp[j] + '>');
            }
        }
    }
    dfs(0, 0, 0);
    for (auto & i : s) {
        cout << "{" << i.second << "}" << endl;
    }
}
